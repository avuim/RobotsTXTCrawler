    document.addEventListener('DOMContentLoaded', function() {
        console.log('DOM Content Loaded');

        const carousel = document.querySelector('.carousel');
        
        if(!carousel) {
            return;
        }

        const prevBtn = document.getElementById('prevBtn');
        const nextBtn = document.getElementById('nextBtn');
        let currentIndex = 0;
        
        const carouselItems = carousel.querySelectorAll('.carousel-item');
        const totalSlides = carouselItems.length;
        console.log('Total slides:', totalSlides);

        prevBtn.addEventListener('click', function () {
            tmEvent('wahretabelle', 'prev', 'sperren');
            currentIndex = (currentIndex - 1 + totalSlides) % totalSlides;
            updateCarousel();
        });

        nextBtn.addEventListener('click', function () {
            tmEvent('wahretabelle', 'next', 'sperren');
            currentIndex = (currentIndex + 1) % totalSlides;
            updateCarousel();
        });

        updateCarousel();

        function updateCarousel() {
            const itemWidth = carouselItems[currentIndex].getBoundingClientRect().width;
            const offset = -currentIndex * itemWidth;
            console.log(`currentIndex: ${currentIndex}, itemWidth: ${itemWidth}, offset: ${offset}`);
            carousel.style.transform = `translateX(${offset}px)`;
        }
    });
