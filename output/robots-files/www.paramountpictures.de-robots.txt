            headerVideo.pause()

        } else {

        }
    });

        currentVideo++;

    checkScreen();

function loadVideo() {
    $.ajax({
        type: "GET", url: "/json/start/trailer", async: true, cache: false, success: function (a) {
            $.each(a, function (b, c) {
                headerAlias.push(c.alias);
                headerPosterMobile.push(media_url + c.poster_mobile);
                headerPosterDesktop.push(media_url + c.poster_desktop);
                headerVideos.push(media_url + "/movies/" + c.id + "/trailer.mp4");
                headerTitles.push(c.title);
                headerSubtitles.push(c.startDate);
                i++
            });
            getCarousel();
            initHeader()
        }, error: function (a) {
            console.log(a.message)
        }
    })
}
function getCarousel() {
    $.get("/json/start/carousel", function (a) {
        $.each(a, function (c, e) {
            var d = e.item;
            $("#slickSlider").append(d)
        });
        var b = "100%";
        if ($(window).width() > 768) {
            b = ($(".container").width() / 5 - 5)
        } else {
            b = "220"
        }
        $(".slickSlider").bxSlider({
            responsive: true,
            minSlides: 2,
            maxSlides: 5,
            slideWidth: b,
            slideMargin: 5,
            pager: false,
            preloadImages: "all",
            nextText: "",
            prevText: "",
            onSliderLoad: function () {
            }
        })
    })
}
function initHeader() {
    headerVideo = document.getElementById("header-video");
    headerVideo.muted = true;
    headerVideo.onended = function (b) {
        currentVideo++;
        if (currentVideo > headerVideos.length - 1) {
            currentVideo = 0
        }
        loadSource(currentVideo);
        if (isTablet) {
            $("#header-controls").fadeIn(1500);
            $("#header-infos").fadeIn(1500);
            $("#header-video-paginator").fadeIn(1500)
        }
    };
    loadSource(currentVideo);
    resizeHeader();
    if (isLandscape) {
        $(".tablet-video-still").css("background-image", "url(" + headerPosterDesktop[currentVideo] + ")")
    } else {
        $(".tablet-video-still").css("background-image", "url(" + headerPosterDesktop[currentVideo] + ")")
    }
    if (isIphone) {
        $(".tablet-video-still").fadeIn()
    }
    for (var a = 0; a < headerVideos.length; a++) {
        $("#header-video-paginator").append("<li><p>&#9679;</p></li>")
    }
    $("#header-video-paginator > li").click(function (b) {
        if ($(this).index() !== currentVideo) {
            currentVideo = $(this).index();
            loadSource(currentVideo);
            if (isLandscape) {
                $(".tablet-video-still").css("background-image", "url(" + headerPosterDesktop[currentVideo] + ")")
            } else {
                $(".tablet-video-still").css("background-image", "url(" + headerPosterDesktop[currentVideo] + ")")
            }
            if (isIphone) {
                $(".tablet-video-still").fadeIn()
            }
            headerVideo.play();
            $("#header-play-button").fadeOut()
        }
    });
    $(window).scroll(function () {
        if ($("#header-video").is(":in-viewport( 800 )")) {
            headerVideo.play()
        } else {
            headerVideo.pause()
        }
    })
}
function setStill() {
    if (!isLandscape) {
        $("#header-video-sound").hide();
        $(".tablet-video-still").css("background-image", "url(" + headerPosterDesktop[currentVideo] + ")");
        $(".tablet-video-still").fadeIn();
    } else {
        if (isIphone || isTablet) {
            $("#header-video-sound").hide()
        } else {
            $("#header-video-sound").show()
        }
    }
    if (isLandscape && (isTablet || isIphone)) {
        $(".tablet-video-still").css("background-image", "url(" + headerPosterDesktop[currentVideo] + ")")
    } else {
        $(".tablet-video-still").css("background-image", "url(" + headerPosterDesktop[currentVideo] + ")")
    }
}
function resizeHeader() {
    setStill();
    if ($(window).width() < 769) {
        $("#header-play-button").fadeOut();
        headerVideo.pause();
        $("#header-video").css("display", "none");
        if (isLandscape) {
            $("#header-content").height(height / width * $(window).width());
            if ($("#header-content").height() > $(window).height()) {
                $("#header-content").height($(window).height())
            } else {
                $("#header-content").height(height / width * $(window).width())
            }
        } else {
            //$("#header-content").height($(window).height())
            $("#header-content").height(height / width * $(window).width())
        }
    } else {
        if (!isIphone) {
            $("#header-video").css("display", "block");
            $("#header-video").width("100%");
            $("#header-video").height(height / width * $("#header-video").innerWidth());
            if (fitHeader && $("#header-video").innerHeight() < $(window).height()) {
                $("#header-content").height($("#header-video").innerHeight())
            } else {
                if (fitHeader) {
                    $("#header-content").height($(window).height())
                }
            }
            if (isDesktop) {
                if (headerVideo.paused) {
                    headerVideo.play()
                }
            }
            if (isTablet) {
                if (!headerVideo.playing) {
                    headerVideo.pause();
                    $("#header-play-button").addClass("animated zoomIn");
                    $("#header-play-button").fadeIn();
                    $(".tablet-video-still").fadeIn()
                }
            }
        }
    }
}
function setHeaderInfos(a) {
    if (headerTitles[a] && headerSubtitles[a]) {
        $("#header-video-title").html('<a href="/' + headerAlias[a] + '">' + headerTitles[a] + "</a>");
        $("#header-video-subtitle").html('<a href="/' + headerAlias[a] + '">' + headerSubtitles[a] + "</a>");
        $("#header-infos").fadeIn(1000)
    }
}
function loadSource(a) {
    var b = document.createElement("source");
    $(b).attr("type", "video/mp4");
    $(b).attr("src", headerVideos[a]);
    $("#header-video").empty();
    $("#header-video").append(b);
    headerVideo.load();
    if (isDesktop) {
        headerVideo.play()
    }
    setHeaderInfos(a)
}

function soundIcon(c) {
    var d = ["-68px -48px", "-68px -25px"];
    if (!c) {
        $("#header-video-sound").css("background-position", d[1])
    } else {
        $("#header-video-sound").css("background-position", d[0])
    }
}
function soundClicked() {
    if (headerVideo.muted) {
        headerVideo.muted = false;
        soundIcon(true)
    } else {
        headerVideo.muted = true;
        soundIcon(false)
    }
}
function checkScreen() {
    checkOrientation();
    if (isMobile.phone) {
        isDesktop = false;
        isTablet = false;
        isIphone = true
    } else {
        if (isMobile.tablet) {
            isDesktop = false;
            isTablet = true;
            isIphone = false
        } else {
            isIphone = false;
            isTablet = false;
            isDesktop = true
        }
    }
}
function checkOrientation() {
    if ($(window).width() > $(window).height()) {
        isLandscape = true;
        isPortrait = false
    } else {
        isLandscape = false;
        isPortrait = true
    }
};

        $("#header-play-button").fadeIn();
